function checkDelivery(total) {
  if (total >= 50000) {
    return "🎉 Free delivery!";
  } else {
    const gap = 50000 - total;
    return "Add " + gap.toLocaleString() + " XAF more for free delivery.";
  }
}

console.log(checkDelivery(112000));  // Free
console.log(checkDelivery(30000));   // Not yet
