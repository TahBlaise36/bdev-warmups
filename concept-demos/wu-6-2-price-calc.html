const price = 112000;      // Tecno Spark 20 Pro (XAF)
const discount = 0.10;    // 10% promo
const saving = price * discount;
const finalPrice = price - saving;

console.log("Original:", price, "XAF");
console.log("You save:", saving, "XAF");
console.log("Final:   ", finalPrice, "XAF");
